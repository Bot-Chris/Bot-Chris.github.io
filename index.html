                            const startIndex = (page - 1) * limit;
                            const endIndex = page * limit;
                            const paginatedCocktails = cocktails.slice(startIndex, endIndex);

                            return res.status(500).json({ error: 'Failed to fetch data from The Cocktail DB' });
                        }
                    });

// API endpoint for creating a new beer gift
router.post('/cheers', authenticateToken, async (req, res) => {
  try {
    const { recipientId, message, beerId, venueId } = req.body;
    const senderId = req.user.id;
    
    // Validate input
    if (!recipientId || !beerId || !venueId) {
      return res.status(400).json({ error: 'Missing required fields' });
    }
    
    // Create transaction record
    const result = await db.query(
      `INSERT INTO beer_gifts (sender_id, recipient_id, beer_id, venue_id, message, sent_date) 
       VALUES (@senderId, @recipientId, @beerId, @venueId, @message, GETDATE());
       SELECT SCOPE_IDENTITY() AS id;`,
      {
        senderId, recipientId, beerId, venueId, 
        message: message || null
      }
    );
    
    return res.status(201).json({ 
      success: true, 
      giftId: result.recordset[0].id 
    });
  } catch (error) {
    console.error('Error creating beer gift:', error);
    return res.status(500).json({ error: 'Internal server error' });
  }
});
